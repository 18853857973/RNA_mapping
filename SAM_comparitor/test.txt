#!/usr/bin/env python

import sys

sam1 = sys.argv[1]
sam2 = sys.argv[2]

output = {sam1: {}, sam2: {}}

#Check genomes are the same
f1= open(sam1)
f2= open(sam2)


leng_1 = 0
leng_2 = 0
for line_1 in f1:
	if "@SQ" in line_1:
		seq1_length = line_1.split()[2]
		leng_1 = int(seq1_length.split(":")[1])
		break
for line_2 in f2:
	if "@SQ" in line_2:
		seq2_length = line_2.split()[2]
		leng_2 = int(seq2_length.split(":")[1])
		break
if  leng_1 != leng_2:
	print("These SAM files appear to be from different references")
	quit()
else:
	print("Both SAM files appear to have used the same reference for alignment")

#Check SAM files are sorted by read name

#Parse map/unmapped reads in each file.
print "Percent unmapped:"
for F in [sam1, sam2]:
	with open(F) as f:
		unmapped = 0
		count    = 0
		primary  = 0

		while True:
			line = f.readline()
			if line == "":
				break
			if not line.startswith("@"):
				flag = int(line.split()[1])
				if flag & 4:
					unmapped += 1
				if not flag & 256:
					primary += 1
				count += 1
	
	pcnt_unmapped = float(unmapped)/primary
	print F + ":\t%f %d/%d" % (pcnt_unmapped, unmapped, primary)
	output[F]["count"] = count
	output[F]["unmapped"] = unmapped
	output[F]["primary"] = primary
